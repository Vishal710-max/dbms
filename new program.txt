/ --Creating Package 
Create OR Replace Package Body math_pow IS 
Procedure cal_squ(n IN Number) IS 
Begin 
dbms_output.put_line('Square of ' || n || ' is: ' || (n * n));

2) Create package which contain procedure, function, and cursor for making 
student marks sheet(use std ,marks table).  
SQL> --Defining Package 
Create OR Replace Package std_pkg IS 
Cursor std_cur(n Number) Return std%ROWTYPE; 
Procedure std_dis(n IN Number); 
Function std_cal(n Number) Return Number; 
End std_pkg; 
/ --Creating Package 
Create OR Replace Package Body std_pkg IS 
Cursor std_cur(n Number) Return std%ROWTYPE IS  
Select * from std Where Roll_No = n; 
Procedure std_dis(n IN Number) IS 
info std%ROWTYPE; 
Begin 
Open std_cur(n); 
Loop 
Fetch std_cur INTO info; 
Exit When std_cur%NOTFOUND; 
dbms_output.put_line('Roll No: ' || info.Roll_No); 
dbms_output.put_line('Name: ' || info.name); 
End Loop; 
Close std_cur; 
End std_dis; 
Function std_cal(n Number) Return Number IS 
m1 Number; 
m2 Number; 
m3 Number; 
per Number; 
Begin 
Select sub1, sub2, sub3 into m1, m2, m3 from marks Where Roll_No = n; 
per := (m1 + m2 + m3) / 3; 
Return trunc(per, 2); 
End std_cal; 
End std_pkg; 
/

1. Package Specification(Defining):- 
CREATE OR REPLACE PACKAGE Package_Name IS -- Cursor Declaration 
CURSOR Cursor_Name(Var_Name Data_Type) RETURN Table_Name%ROWTYPE; -- Procedure Declaration 
PROCEDURE Procedure_Name(Var_Name IN/OUT/IN OUT Data_Type);

-- Function Declaration

END Package_Name; 
/ 
2. Package Body (Implementation):- 
CREATE OR REPLACE PACKAGE BODY Package_Name IS -- Cursor Definition 
CURSOR Cursor_Name(Var_Name Data_Type) RETURN Table_Name%ROWTYPE IS 
SELECT * FROM Table_Name WHERE Col_Name = Var_Name; -- Procedure Implementation 
PROCEDURE Procedure_Name(Var_Name IN/OUT/IN OUT Data_Type) IS 
v_row Table_Name%ROWTYPE; 
BEGIN 
OPEN Cursor_Name(Var_Name); 
LOOP 
FETCH Cursor_Name INTO v_row; 
EXIT WHEN Cursor_Name%NOTFOUND; 
DBMS_OUTPUT.PUT_LINE(v_row.Col_Name1 || ' ' || v_row.Col_Name2); 
END LOOP; 
CLOSE Cursor_Name; 
END Procedure_Name; -- Alternative way using FOR loop
