A trigger in DBMS is a mechanism that automatically performs a defined action in response to specific changes in a database (such as INSERT, UPDATE, or DELETE). 
Triggers can be defined to run before or after the event occurs, depending on the need.
It helps maintain data integrity, automate tasks, and enforce business rules.

CREATE TABLE Employees (
    EmployeeID INT PRIMARY KEY,
    Name VARCHAR(50),
    Salary DECIMAL(10, 2)
);

CREATE TABLE AuditLog (
    LogID INT PRIMARY KEY AUTO_INCREMENT,
    Action VARCHAR(50),
    EmployeeID INT,
    ChangeDate DATETIME
);

CREATE TRIGGER LogEmployeeUpdate
AFTER UPDATE ON Employees
FOR EACH ROW
BEGIN
    INSERT INTO AuditLog (Action, EmployeeID, ChangeDate)
    VALUES ('UPDATE', OLD.EmployeeID, NOW());
END;
